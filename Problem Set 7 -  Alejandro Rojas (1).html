Question 5: Subvector Inference

Add identification robust inference methods for subsets of the parameters.

The current tests are all only for hypotheses involving the entire vector of parameters.

Guggenberger & Smith (2005) have some results for GEL and subvector inference.

Intuition

Starting with the model $u_{ij} = x_j \theta + \xi_j + \epsilon_{ij}$ where $x_j$ are observed variables, $xi_j$ are latent variables, and $\epsilon_{ij}$ are ideosyncratic shocks in preferences about the option $j$ we have that the percentage of individuals that choose $j$ are: $y_j = F_{-\epsilon}\left(x_j \theta + \xi_j\right)$ where $F_{-\epsilon}$ is the CDF of $-\epsilon$.

In the model we assume that $\epsilon_{ij}$ is independent of $x_j$ and $\xi_j$. But we allow $x_j$ to be correlated with $\xi_j$. Let's assume that we have an instrument that satisfies $E\left[\xi_j z_j\right]=0$. Therefore, the moment condition in terms of observables and $\theta$ is given by: $g(x_j, z_j, \theta) = E\left[\left(F_{-\epsilon}^{-1}\left(y_j\right)-x_j\theta\right)z_j\right].$

The problem of subvector inference that we are going to solve imposes a constrain in $\theta^*=\left[\alpha, \beta\right]$ such that $\alpha = \alpha_0$. This means that the constraint GMM problem that we are going to solve is: $\hat{\beta}\left(\alpha_0\right)=\underset{\beta \in B}{arg max} \hat{Q}_n \left(\theta^*\right)$ where $\hat{Q}_n \left(\theta\right) =\left(\frac{1}{n}\sum_{i=1}^{n}g(x_j, z_j, \theta^*)\right)'\hat{W}\left(\frac{1}{n}\sum_{i=1}^{n}g(x_j, z_j, \theta^*)\right)$ and we are going to make inference about the subvector $\beta\left(\alpha_0\right)$.

function gi_ivshare(β,y,x,z)
    ξ = quantile.(Logistic(),y) .- x*β
    ξ.*z

`ordering` takes a set of unconstrained parameters $\psi$ (the $\beta$'s to estimate), a set of constrained parameters $\alpha$, and an indicator matrix $Ind$ that signals the place in $\theta$ of the elements in $\alpha$ on orders them in the same way that $\theta$

{.julia} function ordering(ψ,Ind,α) @assert sum(Ind) == length(α) countψ = 0; countα = 0; countθ = 0; θ = similar(ψ, length(Ind)); for i in 1:length(Ind) countθ += 1 if Ind[i] == 1 countα += 1 θ[countθ] = α[countα] elseif Ind[i] == 0 countψ += 1 θ[countθ] = ψ[countψ] end end return θ end

`gmmObjsub` establishes the GMM objective function for a set of unconstrained parameters $\psi$, a set of constrained parameters $\alpha$, the indicator matrix of constrains $Ind$, a weight matrix $W$, and a moment condition function $$

~~~~{.julia}
function gmmObjsubvec(ψ,gi,W,Ind,α)
    θ = ordering(ψ,Ind,α)
    g = gi(θ)
    m = mean(g, dims=1)
    return (size(g)[1]*( m*W*m')[1])
end

gmmVar estimates the variance of the parameters $\theta$ for a weight matrix $W$ and a moment condition function $gi$

function gmmVar(θ,gi,W)
    g = gi(θ)
    n = size(g)[1]
    D = ForwardDiff.jacobian(θ -> mean(gi(θ),dims=1),θ)
    Σ = cov(gi(θ))
    1/n*inv(D'*W*D)*(D'*W*Σ*W*D)*inv(D'*W*D)
end

Finally we create a function plot_cr that establishes confidence intervals with Anderson-Rubin test and the Wald statistic

Plots.gr()
function plot_cr(β,θ,V, AR, Ind, α)
    #Extract relevant submatrix of V
    Vsub = fill(0.0, sum(Ind), sum(Ind))
    count_row = 0
    for i ∈ 1:length(Ind)
        if Ind[i] == 1
            count_row += 1
            count_col = 0
            for j ∈ 1:length(Ind)
                if Ind[j] == 1
                    count_col += 1
                    Vsub[count_row,count_col] = V[i,j]
                end
            end
        end
    end
    @assert Vsub ≈ transpose(Vsub)
    lb = β - sqrt.(diag(Vsub))*5
    ub = β + sqrt.(diag(Vsub))*5
    ntest = 1000
    βtest = [rand(length(β)).*(ub-lb) .+ lb for i in 1:ntest]
    arstat = [AR(ordering(βtest[i],Ind,α)) for i in 1:ntest]
    βtest = vcat(βtest'...)
    crit = quantile(Chisq(size(z)[2]), 0.9)
    ψ0 = fill(0.0, length(β))
    count_ψ0 = 0
    for i ∈ 1:length(Ind)
        if Ind[i] == 1
            count_ψ0 += 1
            ψ0[count_ψ0] = β0[i]
        end
    end
    b1 = lb[1]:(ub[1]-lb[1])/100:ub[1]
    b2 = lb[2]:(ub[2]-lb[2])/100:ub[2]
    arf = (a,b) -> cdf(Chisq(size(z)[2]),AR(ordering([a;b],Ind,α)))
    scatter(βtest[:,1],βtest[:,2], group=(arstat.<crit), legend=false,
            markersize=4, markerstrokewidth=0.0, seriesalpha=0.8,
            xlabel="beta_1", ylabel="beta_2")
    scatter!([ψ0[1]], [ψ0[2]], markersize=8)
    b1 = lb[1]:(ub[1]-lb[1])/100:ub[1]
    b2 = lb[2]:(ub[2]-lb[2])/100:ub[2]
    arf = (a,b) -> cdf(Chisq(size(z)[2]),AR(ordering([a;b],Ind,α)))
    contour!(b1,b2,arf, levels = [0.9, 0.95], contour_labels=false, legend=false, label="AR CI", c=cgrad([:black,:black],[0.0,1.0]))
    waldf = (a,b) -> cdf(Chisq(length(βcue)),([a,b]-βcue)'*inv(Vsub)*([a,b]-βcue))
    contour!(b1,b2,waldf, levels = [0.9, 0.95],          contour_labels=false, label="Wald CI", c=cgrad([:red,:red], [0.0,1.0]), legend=false)

end
